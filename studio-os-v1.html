⚙ Configurar Google Sheets

Cole a URL gerada após publicar o Apps Script.

Como configurar

1. Crie uma planilha Google Sheets

Acesse sheets.google.com, crie uma planilha e anote o ID (na URL entre /d/ e /edit).

2. Abra o Apps Script

Na planilha: Extensões → Apps Script. Apague o código e cole o abaixo.

3. Cole este código

const SHEET_ID = 'SEU_ID_AQUI';
function doPost(e){
  const d=JSON.parse(e.postData.contents),ss=SpreadsheetApp.openById(SHEET_ID);
  ['colaboradores','projetos','alocacoes','propostas','contratos','extContratos'].forEach(k=>{if(d[k])salvar(ss,k,d[k]);});
  return out({ok:true});
}
function doGet(){
  const ss=SpreadsheetApp.openById(SHEET_ID);
  const keys=['colaboradores','projetos','alocacoes','propostas','contratos','extContratos'];
  const r={};keys.forEach(k=>r[k]=ler(ss,k));return out(r);
}
function salvar(ss,nome,list){
  let a=ss.getSheetByName(nome)||ss.insertSheet(nome);a.clearContents();
  if(!list.length)return;const k=Object.keys(list[0]);a.appendRow(k);
  list.forEach(o=>a.appendRow(k.map(c=>{const v=o[c];return typeof v==='object'&&v?JSON.stringify(v):v??'';})));
}
function ler(ss,nome){
  const a=ss.getSheetByName(nome);if(!a)return[];
  const r=a.getDataRange().getValues();if(r.length<2)return[];
  const k=r[0];return r.slice(1).map(row=>{const o={};k.forEach((c,i)=>{try{o[c]=JSON.parse(row[i]);}catch{o[c]=row[i];}});return o;});
}
function out(d){return ContentService.createTextOutput(JSON.stringify(d)).setMimeType(ContentService.MimeType.JSON);}

4. Publique como Web App

Implantar → Nova implantação → tipo App da Web → acesso Qualquer pessoa → Implantar. Copie a URL e cole acima.

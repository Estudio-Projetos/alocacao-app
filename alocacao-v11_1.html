⚙️ Configurar Google Sheets

Cole aqui a URL gerada após publicar o Apps Script abaixo.

📋 Como configurar — passo a passo

1. Crie a planilha

Acesse sheets.google.com e crie uma planilha nova. Anote o ID dela (está na URL entre /d/ e /edit).

2. Abra o Apps Script

Na planilha, clique em Extensões → Apps Script. Apague o código existente e cole o código abaixo.

3. Cole este código no Apps Script

const SHEET_ID = 'COLE_O_ID_DA_SUA_PLANILHA_AQUI';

function doPost(e) {
  try {
    const dados = JSON.parse(e.postData.contents);
    const ss = SpreadsheetApp.openById(SHEET_ID);
    salvarAba(ss, 'colaboradores', dados.colaboradores);
    salvarAba(ss, 'projetos',      dados.projetos);
    salvarAba(ss, 'alocacoes',     dados.alocacoes);
    return ContentService.createTextOutput(
      JSON.stringify({ok:true})).setMimeType(ContentService.MimeType.JSON);
  } catch(err) {
    return ContentService.createTextOutput(
      JSON.stringify({ok:false,erro:err.message})).setMimeType(ContentService.MimeType.JSON);
  }
}

function doGet(e) {
  const ss = SpreadsheetApp.openById(SHEET_ID);
  const dados = {
    colaboradores: lerAba(ss,'colaboradores'),
    projetos:      lerAba(ss,'projetos'),
    alocacoes:     lerAba(ss,'alocacoes')
  };
  return ContentService.createTextOutput(
    JSON.stringify(dados)).setMimeType(ContentService.MimeType.JSON);
}

function salvarAba(ss, nome, lista) {
  let aba = ss.getSheetByName(nome);
  if (!aba) aba = ss.insertSheet(nome);
  aba.clearContents();
  if (!lista || lista.length === 0) return;
  const chaves = Object.keys(lista[0]);
  aba.appendRow(chaves);
  lista.forEach(obj => aba.appendRow(chaves.map(k => {
    const v = obj[k];
    return (typeof v === 'object' && v !== null) ? JSON.stringify(v) : (v ?? '');
  })));
}

function lerAba(ss, nome) {
  const aba = ss.getSheetByName(nome);
  if (!aba) return [];
  const rows = aba.getDataRange().getValues();
  if (rows.length < 2) return [];
  const chaves = rows[0];
  return rows.slice(1).map(r => {
    const obj = {};
    chaves.forEach((k,i) => {
      const v = r[i];
      try { obj[k] = JSON.parse(v); } catch { obj[k] = v; }
    });
    return obj;
  });
}

4. Publique como Web App

Clique em Implantar → Nova implantação. Tipo: App da Web. Em "Quem tem acesso" selecione Qualquer pessoa. Clique em Implantar e copie a URL gerada.

5. Cole a URL acima e salve

Cole a URL no campo acima e clique em Salvar configuração. Depois use ☁️ Salvar no Sheets para enviar os dados e 🔄 Carregar do Sheets para buscar.
